#include< iostream>
    #include<bits/stdc++.h>
    using namespace std;


    int main ()
    {
    
        unordered_map<int , bool> array_table;
        int X[] = { 3, 5, -1, 8, 11, 1, -1, 6 };
        int target = 13;
        int size = 8;
        int Y = 0;
    
        for(int i=0; i<size; i++){
            Y = target - X[i];
    
            if( array_table[Y] ){
        
            cout<< Y << " , " << X[i] <<endl;
    
            }
            else{
    
                array_table[ X[i] ] = true;
    
            }
        }
        return 0;
    }
