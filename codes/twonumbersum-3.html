#include< iostream>
    #include<bits/stdc++.h>
    using namespace std;


    int main ()
    {
    
        
        int arr[] = { 3, 5, -1, 8, 11, 1, -1, 6 };
        int target = 13;
        int size = 8;
        int sum = 0;
        int L = 0, R = size - 1;
    
        //Sorting array
        sort(arr,arr + size);
        
        while( (arr[L] + arr[R] )!= target && L < size && R > -1 ){

            if( (arr[L] + arr[R] ) < target ){
                R --;
            }
            else{
                L++;
            }
        cout<< arr[L] << " , " << arr[R] <<endl;
        return 0;
    }
