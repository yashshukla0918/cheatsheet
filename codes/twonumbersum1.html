using namespace std;
int * getPair (int arr[], int size, int target)
{

    static int ans[2];
    for (int i = 0; i < size; i++)
    {

        for (int j = 0; j < size; j++)
    {

        if (target == (arr[i] + arr[j]))
        {

            ans[0] = arr[i];
            ans[1] = arr[j];

            return ans;
        }
    }
    }
}



int main ()
{

    int arr[] = { 3, 5, -1, 8, 11, 1, -1, 6 };
    int target = 13;
    int size = 8;
    int *p;

    p = getPair (arr, size, target);

    for (int i = 0; i < 2; i++)
    {

        cout << *(p + i) << " ";

    }
    cout << endl;

    return 0;
}
